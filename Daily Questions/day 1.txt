FUNCTION validateAndSummarizeWorkSessions(SESSION)
ERROR=[]
SESSIONCOUNT=0
TOTALMINUTEWORK=0

IF SESSION IS EMPTY THEN
    RETURN {ISVALID=FALSE,ERROR MESSAGE:"SESSION IS EMPTY"}
END IF

FOR I START FROM 0 LENGTH(SESSION)-1
SORT SESSION BY STARTTIME
IF STARTTIME or ENDTIME is NULL
    ADD ERROR "Missing time in session i"
IF(SESSION[I] STARTTIME >= SESSION[I] ENDTIME) THEN 
    ADD ERROR[] MESSAGE:"SESSION I HAS INVALID START AND ENDTIME"
ELSE IF(SESSION[I] STARTTIME <0 OR SESSION[I] ENDTIME >1439) THEN 
    ADD ERROR[] MESSAGE:"NOT IN SAME DAY"
ELSE
DURATION = SESSION[I] ENDTIME - SESSION[I] STARTTIME
   TOTALMINUTEWORK=TOTALMINUTEWORK+DURATION
    SESSIONCOUNT=SESSIONCOUNT+1
END IF

IF(I < LENGTH(SESSION)-1) THEN
IF(SESSION[I] ENDTIME > SESSION[I+1] STARTTIME AND SESSION[I+1] ENDTIME > SESSION[I] STARTTIME) THEN
    ADD ERROR[] MESSAGE:"THE SESSIONS ARE OVERLAPPING"
END IF
END IF

END FOR

IF ERROR IS EMPTY THEN
    RETURN 
    {ISVALID=TRUE,
    TOTALMINUTEWORK=TOTALMINUTEWORK,
    SESSIONCOUNT=SESSIONCOUNT}
ELSE
    RETURN 
    {ISVALID=FALSE,
    ERROR MESSAGE:ERROR}
END IF

END FUNCTION
