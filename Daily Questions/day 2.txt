->FIRST START FROM THE CART SUBTOTAL AMOUNT
->ADD SHIPPING COST
    BASED ON THE SHIPPING METHOG(NORMAL OR FAST)
->APPLY DISCOUNT
    PERCENTAGE OR FIXED COST
->CALCULATE TAX
    BASE ON THE DESTINATION
    TAX APPLIED AFTER THE DISCOUNT
->COMPLETE THE FINAL TOTAL
    IF TOTAL < 0 MAKE IT O


ENTER THE CART PAGE AND GIVE :
INPUT CARTSUBTOTAL(TOTAL PRODCT PRICE IN CART)
CHOOSE INPUT SHIPPINGMETHOD:
IF SHIPPINGMETHOD = "NORMAL" THEN
    SHIPPINGCOST = 50
ELSE IF SHIPPINGMETHOD = "FAST" THEN
    SHIPPINGCOST = 100
END IF
AMOUNTAFTERSHIPPING = CARTSUBTOTAL + SHIPPINGCOST
GIVE INPUT DISCOUNTTYPE   
GIVE INPUT DISCOUNTVALUE
IF DISCOUNTTYPE = "PERCENTAGE" THEN
    DISCOUNTAMOUNT = (AMOUNTAFTERSHIPPING * DISCOUNTVALUE) / 100
ELSE IF DISCOUNTTYPE = "FIXED" THEN
    DISCOUNTAMOUNT = DISCOUNTVALUE
END IF
AMOUNTAFTERDISCOUNT = AMOUNTAFTERSHIPPING - DISCOUNTAMOUNT
GIVE INPUT DESTINATION
IF DESTINATION = "LOCAL" THEN
TAX = 5%
ELSE IF DESTINATION = "OUTSTATION" THEN
TAX = 12%
ELSE
TAX = 18%
END IF
TAXAMOUNT = AMOUNTAFTERDISCOUNT * TAX / 100
FINALAMOUNT = AMOUNTAFTERDISCOUNT + TAXAMOUNT
IF FINALAMOUNT < 0 THEN
FINALAMOUNT = 0
END IF
DISPLAY FINALAMOUNT
